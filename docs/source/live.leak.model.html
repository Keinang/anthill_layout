define([
    'modules/Model',
    'plugins/widgets/widget.content.model'
], function defineLiveLeakModel(BaseModel, WidgetContentModel) {

    /**
     * Define LiveLeak model
     * @extends BaseModel
     * @extends WidgetContentModel
     * @class LiveLeakModel
     * @constructor
     */
    var LiveLeakModel = function LiveLeakModel() {

        /**
         * Define preferences
         * @member LiveLeakModel
         * @type {{
         *      liveleakUrl: {type: string, disabled: boolean, value: undefined, visible: boolean}
         * }}
         */
        this.preferences = {
            liveleakUrl: {
                type: 'text',
                disabled: false,
                value: undefined,
                visible: true
            }
        };

        /**
         * Define rules
         * @member LiveLeakModel
         * @type {{}}
         */
        this.rules = {};
    };

    return LiveLeakModel.extend('LiveLeakModel', {

        /**
         * Set LiveLeak Url
         * @member LiveLeakModel
         * @param {string} url
         */
        setLiveleakUrl: function setLiveleakUrl(url) {
            this.setPrefs('liveleakUrl', url);
        }

    }, BaseModel.prototype, WidgetContentModel.prototype);
});
