define([
    'modules/base',
    'modules/mvc',
    'api/template.api',
    'config/layout',
    'controller/template.controller',
    'model/template.model',
    'view/template.view',
    'event/template.event.manager'
], function defineTemplate(Base, MVC, API, Layout, Controller, Model, View, EventManager) {

    /**
     * Define template
     * @class Template
     * @extends {Base}
     * @param opts
     * @param page
     * @constructor
     */
    var Template = function Template(opts, page) {

        opts = this.base.define(opts, {}, true);

        /**
         * Default config
         * @type {{parent: *}}
         */
        var DEFAULTS = {
            parent: page
        };

        /**
         * Define MVC
         * @type {MVC}
         */
        this.mvc = new MVC({
            scope: this,
            config: [opts, DEFAULTS],
            components: [
                API,
                Controller,
                Model,
                View,
                EventManager
            ],
            render: true
        });

        /**
         * Define page
         * @type {*|Page}
         */
        this.page = {};

        /**
         * Define items
         * @type {*}
         */
        this.items = {};

        this.observer.publish(this.eventmanager.eventList.successCreated);

    };

    return Template.extend(Base);
});
