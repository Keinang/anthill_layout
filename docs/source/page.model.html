define([
    'modules/model',
    'modules/base',
    'config/widget'
], function definePageModel(BaseModel, Base, Widget) {
    var Model = function Model() {
        this.item = Widget;
        this.onDestroy = [
            'Layout',
            'Template'
        ];
    };

    return Model.extend({

    }, BaseModel.prototype, Base);
});
