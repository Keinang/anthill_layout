define([
    'modules/Model',
    'plugins/widgets/widget.content.model'
], function defineOpenWeatherMapModel(BaseModel, WidgetContentModel) {

     * @extends WidgetContentModel
     * @class OpenWeatherMapModel
     * @constructor
     */
    var OpenWeatherMapModel = function OpenWeatherMapModel() {

        /**
         * Define preferences

         */
        this.preferences = {
            openweathermapApiKey: {
                type: 'text',
                disabled: true,
                value: '259681e7bb4cfa2cd775fb127d09ca74',

                disabled: false,

                visible: true
            },
            openweathermapZoom: {
                type: 'text',
                disabled: false,
                value: 14,
                visible: true
            },
            openweathermapWidth: {
                type: 'text',
                disabled: false,
                value: 400,
                visible: true
            },
            openweathermapHeight: {
                type: 'text',
                disabled: false,
                value: 300,
                visible: true
            },

                type: 'combobox',

            }
        };

        /**
         * Define rules

         * @member OpenWeatherMapModel
         * @param {number} latitude

        setOpenWeatherMapLatitude: function setOpenWeatherMapLatitude(latitude) {

         * @param {boolean} sensor
         */

         * @param {boolean} stretch
